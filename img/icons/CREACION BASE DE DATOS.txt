USE ARIMANABAZAR
GO

CREATE TABLE TIPO_USUARIO(
IdTipoUsu INT PRIMARY KEY IDENTITY (1,1),
NombreTipo VARCHAR(50) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)


CREATE TABLE DEPARTAMENTO(
IdDepartamento INT PRIMARY KEY IDENTITY (1,1),
NombreDepa VARCHAR(40) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)


CREATE TABLE PROVINCIA(
IdProvincia INT PRIMARY KEY IDENTITY (1,1),
IdDepartamento INT NOT NULL,
NombreProvincia VARCHAR(40) NOT NULL,
Estado CHAR(1)  NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdDepartamento) REFERENCES DEPARTAMENTO(IdDepartamento)
)


CREATE TABLE DISTRITO(
IdDistrito INT PRIMARY KEY IDENTITY (1,1),
IdProvincia INT NOT NULL,
NombreDistrito VARCHAR(40) NOT NULL,
Estado CHAR(1)  NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdProvincia) REFERENCES PROVINCIA(IdProvincia)
)

CREATE TABLE PERSONA(
IdPersona INT PRIMARY KEY IDENTITY (1,1),
IdDIstrito INT NOT NULL,
Nombres VARCHAR(50) NOT NULL,
ApePaterno VARCHAR(20) NOT NULL,
ApeMaterno VARCHAR(20) NOT NULL,
Dni CHAR(8)  NOT NULL CHECK (Dni LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
Telefono CHAR(9)  NOT NULL UNIQUE CHECK (Telefono LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
GENERO CHAR(1) NOT NULL CHECK(GENERO IN('M', 'F')),
Direccion VARCHAR(60)  NOT NULL,
CORREO VARCHAR(100) DEFAULT('SIN_CORREO'),
FechaNac DATE  NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]') 
FOREIGN KEY (IdDistrito) REFERENCES DISTRITO(IdDistrito)
)


CREATE TABLE CARGO(
IdCargo INT PRIMARY KEY IDENTITY (1,1),
NombreCargo VARCHAR(40) NOT NULL,
Descripcion VARCHAR(10) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)


CREATE TABLE CONTRATO(
IdContrato INT PRIMARY KEY IDENTITY (1,1),
NombreContrato VARCHAR(40) NOT NULL,
Fecha DATE NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)


CREATE TABLE EMPLEADO(
IdEmpleado INT PRIMARY KEY IDENTITY (1,1),
IdCargo INT NOT NULL,
IdContrato INT NOT NULL,
IdPersona INT NOT NULL,
Salario DECIMAL(6,2) NOT NULL,
Turno CHAR(1) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdCargo) REFERENCES CARGO(IdCargo),
FOREIGN KEY (IdContrato) REFERENCES CONTRATO(IdContrato),
FOREIGN KEY (IdPersona) REFERENCES PERSONA(IdPersona)
)


CREATE TABLE ASISTENCIA(
IdAsistencia INT PRIMARY KEY IDENTITY (1,1),
IdEmpleado INT NOT NULL,
Fecha DATE NOT NULL,
Entrada DATETIME NOT NULL,
Salida DATETIME NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdEmpleado) REFERENCES EMPLEADO(IdEmpleado)
)


CREATE TABLE USUARIO(
IdUsuario INT PRIMARY KEY IDENTITY (1,1),
IdEmpleado INT NOT NULL,
IdTipoUsu INT NOT NULL,
Correo VARCHAR(60) NOT NULL,
Contraseña VARCHAR(50) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdTipoUsu) REFERENCES TIPO_USUARIO(IdTipoUsu),
FOREIGN KEY (IdEmpleado) REFERENCES EMPLEADO(IdEmpleado)	
)

CREATE TABLE EMPRESA(
IdEmpresa INT PRIMARY KEY IDENTITY (1,1),
IdDistrito INT NOT NULL,
RazonSocial VARCHAR(50) NOT NULL UNIQUE,
Ruc CHAR(11) NOT NULL UNIQUE CHECK (Ruc LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
Direccion VARCHAR(60) NOT NULL,
Telefono CHAR(9) NOT NULL UNIQUE CHECK (Telefono LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdDistrito) REFERENCES DISTRITO(IdDistrito)
)

CREATE TABLE PROVEDOR(
IdProvedor INT PRIMARY KEY IDENTITY(1,1),
IdEmpresa INT NOT NULL,
Nombre VARCHAR(100) NOT NULL,
Contacto VARCHAR(100) NOT NULL,
Email VARCHAR(150) DEFAULT('SIN_CORREO'),
Telefono CHAR(9) NOT NULL UNIQUE CHECK (Telefono LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
Pagina_web VARCHAR(200) DEFAULT('SIN_PAGINA'),
Descripción VARCHAR(200) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdEmpresa) REFERENCES EMPRESA(IdEmpresa)
)


CREATE TABLE CLIENTE(
IdCliente INT PRIMARY KEY IDENTITY (1,1),
IdPersona INT NOT NULL,
IdEmpresa INT NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdPersona) REFERENCES PERSONA(IdPersona),
FOREIGN KEY (IdEmpresa) REFERENCES EMPRESA(IdEmpresa)	
)


CREATE TABLE TIPO_PAGO(
IdTipoPago INT PRIMARY KEY IDENTITY (1,1),
NomPago VARCHAR(30) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)

CREATE TABLE CATEGORIA(
IdCategoria INT PRIMARY KEY IDENTITY (1,1),
NomCategoria VARCHAR(30) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)

CREATE TABLE PRODUCTO(
IdProducto INT PRIMARY KEY IDENTITY (1,1),
IdCategoria INT NOT NULL,
NomProducto VARCHAR(50) NOT NULL,
Descripcion VARCHAR(100) NOT NULL,
Precio DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdCategoria) REFERENCES CATEGORIA(IdCategoria)
)

CREATE TABLE STOCK_ALMACEN(
IdAlmacen INT PRIMARY KEY IDENTITY (1,1),
IdProducto INT NOT NULL,
NumAlmacen INT NOT NULL,
Stock INT NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY(IdProducto) REFERENCES PRODUCTO(IdProducto)
)

CREATE TABLE PEDIDO(
IdPedido INT PRIMARY KEY IDENTITY (1,1),
Fechapedido DATE NOT NULL,
Total DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
)

CREATE TABLE DETALLE_PEDIDO(
IdDetalleped INT PRIMARY KEY IDENTITY (1,1),
IdPedido INT NOT NULL,
IdProducto INT NOT NULL,
IdUsuario INT NOT NULL,
Cantidad INT NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdPedido) REFERENCES PEDIDO(IdPedido),
FOREIGN KEY (IdProducto) REFERENCES PRODUCTO(IdProducto),
FOREIGN KEY (IdUsuario) REFERENCES USUARIO(IdUsuario)
)

CREATE TABLE VENTA(
IdVenta INT PRIMARY KEY IDENTITY (1,1),
IdPedido INT NOT NULL,
IdTipoPago INT NOT NULL,
IdCliente INT NOT NULL,
Documento CHAR(1) NOT NULL,
Monto DECIMAL(6,2) NOT NULL,
Igv DECIMAL(6,2) NOT NULL,
Subtotal DECIMAL(6,2) NOT NULL,
PagoTotal DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdTipoPago) REFERENCES TIPO_PAGO(IdTipoPago),
FOREIGN KEY (IdPedido) REFERENCES PEDIDO(IdPedido),
FOREIGN KEY (IdCliente) REFERENCES CLIENTE(IdCliente)
)

CREATE TABLE VENTA_SALIDA(
IdVentasalida INT PRIMARY KEY IDENTITY (1,1),
IdCliente INT NOT NULL,
NumFactura CHAR(4) NOT NULL,
Monto DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdCliente) REFERENCES CLIENTE(IdCliente)
) 

CREATE TABLE DETALLE_VENTASALIDA(
IdDetalleVent INT PRIMARY KEY IDENTITY (1,1),
IdVentasalida INT NOT NULL,
IdProducto INT NOT NULL,
Cantidad INT NOT NULL,
Precio DECIMAL(6,2) NOT NULL,
PrecioTotal DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdVentasalida) REFERENCES VENTA_SALIDA(IdVentasalida),
FOREIGN KEY (IdProducto) REFERENCES PRODUCTO(IdProducto),
)


CREATE TABLE FACTURA(
IdFactura INT PRIMARY KEY IDENTITY (1,1),
IdVenta INT NOT NULL,
NumeroCorrelativo CHAR(4) NOT NULL,
Serie CHAR(6) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdVenta) REFERENCES VENTA(IdVenta)
)

CREATE TABLE BOLETA(
IdBoleta INT PRIMARY KEY IDENTITY (1,1),
IdVenta INT NOT NULL,
NumeroCorrelativo CHAR(4) NOT NULL,
Serie CHAR(6) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdVenta) REFERENCES VENTA(IdVenta)
)

CREATE TABLE DEVOLUCION(
IdDevolucion INT PRIMARY KEY IDENTITY (1,1),
IdVenta INT NOT NULL,
FechaDev DATE NOT NULL,
Descripcion VARCHAR(100) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdVenta) REFERENCES VENTA(IdVenta)
)

CREATE TABLE DETALLE_DEVOLUCION(
IdDetalleDev INT PRIMARY KEY IDENTITY (1,1),
IdProducto INT NOT NULL,
IdDevolucion INT NOT NULL,
IdUsuario INT NOT NULL,
Cantidad INT NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdUsuario) REFERENCES USUARIO(IdUsuario),
FOREIGN KEY (IdProducto) REFERENCES PRODUCTO(IdProducto),
FOREIGN KEY (IdDevolucion) REFERENCES DEVOLUCION(IdDevolucion)
)


CREATE TABLE COMPRA_ENTRADA(
IdCompraentrada INT PRIMARY KEY IDENTITY (1,1),
IdProvedor INT NOT NULL,
Documento CHAR(1) NOT NULL,
FechaCompra DATE NOT NULL,
Monto DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdProvedor) REFERENCES PROVEDOR(IdProvedor)
)

CREATE TABLE DETALLE_COMPRA(
IdDetalleCom INT PRIMARY KEY IDENTITY (1,1),
IdCompraentrada INT NOT NULL,
IdProducto INT NOT NULL,
IdUsuario INT NOT NULL,
Cantidad INT NOT NULL,
Precio DECIMAL(6,2) NOT NULL,
PrecioTotal DECIMAL(6,2) NOT NULL,
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdCompraentrada) REFERENCES COMPRA_ENTRADA(IdCompraentrada),
FOREIGN KEY (IdProducto) REFERENCES PRODUCTO(IdProducto),
FOREIGN KEY (IdUsuario) REFERENCES USUARIO(IdUsuario)
)

CREATE TABLE KARDEX(
IdKardex INT PRIMARY KEY IDENTITY (1,1),
IdCompraentrada INT NOT NULL,
IdVentasalida INT NOT NULL,
IdDevolucion INT NOT NULL,
Descripción VARCHAR(100),
Estado CHAR(1) NOT NULL CHECK (Estado LIKE '[0-1]')
FOREIGN KEY (IdCompraentrada) REFERENCES COMPRA_ENTRADA(IdCompraentrada),
FOREIGN KEY (IdVentasalida) REFERENCES VENTA_SALIDA(IdVentasalida),
FOREIGN KEY (IdDevolucion) REFERENCES DEVOLUCION(IdDevolucion)
)
